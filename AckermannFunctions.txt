ll solve(ll x)
{
	ll cnt = 0;
	bool t = false;
	while (!t)
	{
		if (x & 1)
			x = 3 * x + 1;
		else
			x /= 2;
		if (x == 1)
			t = true;
		cnt++;
	}
	return cnt;
}
