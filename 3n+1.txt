ll solve(ll x)
{
	ll cnt = 0;
	while (x>1)
	{
		if (x & 1)
			x = 3 * x + 1;
		else
			x /= 2;
		cnt++;
	}
	return cnt;
}
int main()
{
	fast();
	ll x, y;
	while (scanf("%lld%lld", &x, &y)!=EOF)
	{
		ll mx = -oo;
		
		printf("%lld %lld ", x, y);
		if (y < x)
			swap(x, y);
		for (ll i = x; i <= y; i++)
			mx = max(mx, solve(i));
		printf("%lld\n", mx + 1);
	}
